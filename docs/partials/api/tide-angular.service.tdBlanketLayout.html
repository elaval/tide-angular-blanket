tdBlanketLayout
service in module tide-angular

Description

Helper service to calculate coordinates for set of area shapes that represent an histogram of subjects for different values of the measurement attribute

Example:

angular.module("tide-angular")
.directive("tdBlanket",['tdBlanketLayout',function (tdBlanketLayout) {
  ...

  this.data = [
   {category='A', x:"100", size:"1324"}, 
   {category='A', x:"110", size:"2543"},
   {category='B', x:"100", size:"432"},
   {category='C', x:"110", size:"654"}
  ];

  var layout =  tdBlanketLayout()
    .categoryAttribute("category")
    .xAttribute("x")
    .sizeAttribute("size");

  var nodes = layout.nodes(data);

  ...
}]);

output data format (nodes):

[
 {category:'A', 
  data:[
    {x=100, dy=1324, basey=0},
    {x=110, dy=2543, basey=0}
  ]
 },
 {category:'B', 
  data:[
    {x=100, dy=432, basey=1324},
    {x=110, dy=654, basey=2543}
  ]
 }
]
